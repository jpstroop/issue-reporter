Code used to search for recording FileSets with un-characterized audio files:

class WavFileSets
  def uncharacterized_wav_file_sets
    recordings = query_service.custom_queries.find_by_property(property: "change_set", value: "recording", model: ScannedResource, lazy: false)
    file_sets = recordings.map do |recording|
      recording.decorate.file_sets.select do |file_set|
        file_set.mime_type == ["content/unknown"]
      end
    end.compact.flatten
    file_sets
  end

  def query_service
    @query_service ||= Valkyrie::MetadataAdapter.find(:postgres).query_service
  end
