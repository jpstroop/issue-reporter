Creator:

James R. Griffin III at 2019-10-29 13:57:02 :

This needed to be resolved with the inclusion of the following:

#!/usr/bin/env jruby
require "highline/import"
require "cli"

netid = 'monikam';
name = ask "Collection Name ";
name = name.strip;

require 'dspace'
DSpace.load()
DSpace.login netid

# This is the root collection for all Library Collections
parent_coll = DSpace.fromString('88435/dsp01g732d9019')
template_coll  = DSpace.fromString('88435/dsp018c97kq48z')

puts "Name:\n\t#{name}";
puts "Template Colection:\n\t#{template_coll.getName}\n\tin #{template_coll.getParentObject().getName}";
yes = ask "do you want to create ? [Y/N] "
if (yes[0] == 'Y') then
    options =  {
        netid: netid,
        template_coll: template_coll.getHandle(),
        parent_handle: parent_coll.getHandle(),
        name: name
    };
    puts ["\tTEMPLATE",template_coll].join "\t"
    new_col = DSpace.create(template_coll).copy(name, parent_coll)
    DSpace.commit
    puts "Committed #{new_col.getHandle()}"
    puts "If restricted access: set 'DEFAULT_BITSTREAM_READ' to Princetion_IPs"
    puts 'add to input-forms'
    puts '   <name-map collection-handle="' + new_col.getHandle + '" form-name="digpubs_serials"/>'

end

(ruby/cli/princetonlibrary/createtop_collection.rb)

!/bin/bash

cd ~/ruby/cli jrdspace princetonlibrary/createtopcollection.rb ``` (princetonlibrary/createtop_collection)
