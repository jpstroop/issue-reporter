count = vector_resources.count
i = 0
vector_resources.map do |v|
  i += 1
  puts "#{i}/#{count}"
  gms = v.decorate.geo_members
  if gms.present?
    gms[0].size[0].to_i
  else
    0
  end
rescue
  0
