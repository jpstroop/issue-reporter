#5652 Some FGDC FileSets can't be preserved

Creator:
Trey Pendragon
Created:
2023-01-27 18:26:46
Status:
Closed
Comments:
Eliot Jordan at 2023-05-25 16:38:44 :

Ran script to re-ingest failing fdgc.xml files. Ran cloud fixity check on several and they look good.

ids = []

CSV.foreach("/home/deploy/fgdc.csv") do |row|
  ids << row[0]
end

ids.each do |file_set_id|
  change_set_persister = ChangeSetPersister.default
  query_service = change_set_persister.query_service
  file_set = query_service.find_by(id: file_set_id)
  parent = Wayfinder.for(file_set).parent
  file_path = file_set.primary_file.file_identifiers.first.to_s.gsub("disk://","")
  file = IngestableFile.new(
          file_path: file_path,
          mime_type: "application/xml",
          original_filename: "fgdc.xml",
          copy_before_ingest: true,
          container_attributes: {
            title: "fgdc.xml"
          }
        )
  parent_cs = ChangeSet.for(parent)
  parent_cs.files = [file]
  change_set_persister.save(change_set: parent_cs)
rescue
  next
end

ids.each do |file_set_id|
  change_set_persister = ChangeSetPersister.default
  query_service = change_set_persister.query_service
  file_set = query_service.find_by(id: file_set_id)
  change_set_persister.delete(change_set: ChangeSet.for(file_set))
rescue
  next
end
