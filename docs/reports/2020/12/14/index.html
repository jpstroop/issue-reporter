This is a basic implementation of the Luhn algorithm for computing the checksum (14th digit): def compute_barcode_checksum(barcode_without_checksum) is_odd = true sum_total = 0 while barcode_without_checksum > 0 do last_digit = barcode_without_checksum % 10 #odd digits get multiplied by 2 and summed #if product is => 10, subtract 9 if is_odd then do_odd = last_digit * 2 sum_total = if do_odd < 10 then sum_total + do_odd else sum_total + (do_odd - 9) end # if even, add to sum else sum_total = sum_total + last_digit end is_odd = not(is_odd) # integer divide over 10 to move on to the next digit barcode_without_checksum = barcode_without_checksum / 10 end remainder = sum_total % 10 if remainder == 0 then return remainder else return 10 - remainder end end
