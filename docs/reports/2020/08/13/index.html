    context "when the resource is part of a collection with restricted viewers" do
      let(:restricted_collection) { FactoryBot.create_for_repository(:collection, restricted_viewers: ["zelda"]) }
      let(:resource) do
        change_set = ChangeSet.for(FactoryBot.build(:complete_reading_room_scanned_resource))
        change_set.validate(source_metadata_identifier: component_id, state: ["complete"], member_of_collection_ids: [restricted_collection.id])
        change_set_persister.save(change_set: change_set)
      end
      let(:resource2) { nil }

      it "adds a DAO link for the resource" do
        expect { exporter.export }.not_to raise_error
        after = Nokogiri::XML(File.open(temp_ead))
        expect(after.xpath(xpath, ns).map(&:to_s)).to eq [
          "http://www.example.com/concern/scanned_resources/#{resource.id}/manifest"
        ]
      end
    end
