The following new files needed to be implemented for generating the report:

module Metadata
  class Field
    attr_reader :schema, :element, :qualifier

    def initialize(schema, element, qualifier = nil)
      @schema = schema
      @element = element
      @qualifier = qualifier
    end

    def to_s
      if qualifier.nil?
        "#{schema}.#{element}"
      else
        "#{schema}.#{element}.#{qualifier}"
      end
    end
  end
end

module Reports

  class SeniorThesesReport
    def self.build(handle)
      community = DSpace.fromString(handle)
      self.new(community)
    end

    def initialize(community)
      @community = community
      @rows = []
    end

    def self.kernel
      DSpace
    end

    def self.headers
      ['url', 'ARK', 'title', 'pu.mudd.walkin']
    end

    def self.title_field
      Metadata::Field.new('dc', 'title')
    end

    def self.access_field
      Metadata::Field.new('pu', 'mudd', 'walkin')
    end

    def self.year_field
      Metadata::Field.new('pu', 'date', 'classyear')
    end

    def collections
      @collections ||= begin
                         objects = DCommunity.getCollections(@community)
                         objects.map { |obj| DCollection.new(obj) }
                       end
    end

    def items
      @items ||= collections.items
    end

    def filtered_items
      filtered = []

      items.each do |item|
        class_years = item.getMetadataByMetadataString(self.class.year_field.to_s).collect {|v| v.value}
        class_year = class_years.first

        access_values = item.getMetadataByMetadataString(self.class.access_field.to_s).collect {|v| v.value}
        access = access_values.first

        if class_year == '2020' && access =~ /yes/i
          filtered << item
        end
      end

      filtered
    end

    def generate
      filtered_items.each do |item|
        handle = item.getHandle
        url = "https://dataspace-dev.princeton.edu/handle/#{handle}"

        titles = item.getMetadataByMetadataString(self.class.title_field.to_s).collect {|v| v.value}
        title = titles.first

        access_values = item.getMetadataByMetadataString(self.class.access_field.to_s).collect {|v| v.value}
        access = access_values.first

        row = [url, handle, title, 'pu.mudd.walkin']
        @rows << row
      end

      @rows
    end
  end
end
