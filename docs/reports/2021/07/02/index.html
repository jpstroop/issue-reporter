This might be best structured in the following manner:


group_names = [
  "Lib_DigPubs_Submitters",
  "Lib_DigPubs_Reviewers"
]

def perform(group_name)
  group = org.dspace.eperson.Group.findByName(DSpace.context, group_name)
  members = org.dspace.eperson.Group.allMembers(DSpace.context, group)

  items = []
  members.to_a.each do |member|
    objects = org.dspace.workflow.WorkflowItem.findByEPerson(DSpace.context, member)
    items << objects
  end

  members.to_a.each do |member|
    objects = org.dspace.workflow.WorkflowItem.findByEPerson(DSpace.context, member)
    delta = items.uniq - objects
    workflows = objects.to_a.map { |obj| DSpace::CLI::WorkflowItem.new(obj) }
    workflows.each { |workflow| workflow.add_task_pool_user(member.getEmail) }
  end
end

group_names.each { |group_name| perform(group_name) }
